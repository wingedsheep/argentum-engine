Card Development Guide

How to implement new Magic cards using the cardDef DSL. Patterns, examples, and best practices.

Workflow

  1. Create file in mtg-sets/src/.../definitions/{set}/cards/CardName.kt
  2. Define card using cardDef { } DSL
  3. Add to set definition in definitions/{set}/YourSet.kt
  4. Write scenario tests for interesting mechanics
  5. Engine loads cards automatically via ServiceLoader

Simple Instant

val LightningBolt = cardDef("Lightning Bolt") {
    manaCost = "{R}"
    typeLine = "Instant"

    spell {
        val any = target("any", Targets.Any)
        effect = Effects.DealDamage(3, any)
    }

    metadata {
        rarity = Rarity.COMMON
    }
}

Vanilla Creature

val GlorySeeker = cardDef("Glory Seeker") {
    manaCost = "{1}{W}"
    typeLine = "Creature — Human Soldier"
    power = 2
    toughness = 2
}

Creature with Keywords

val AlabasterDragon = cardDef("Alabaster Dragon") {
    manaCost = "{4}{W}{W}"
    typeLine = "Creature — Dragon"
    power = 4; toughness = 4
    keywords(Keyword.FLYING)

    triggeredAbility {
        trigger = Triggers.Dies
        effect = Effects.PutOnTopOfLibrary(EffectTarget.Self)
    }
}

ETB Triggered Ability

val Nekrataal = cardDef("Nekrataal") {
    manaCost = "{2}{B}{B}"
    typeLine = "Creature — Human Assassin"
    power = 2; toughness = 1
    keywords(Keyword.FIRST_STRIKE)

    triggeredAbility {
        trigger = Triggers.EntersBattlefield
        val t = target("creature", Targets.NonblackNonartifactCreature)
        effect = Effects.Destroy(t)
    }
}

Activated Ability

val Sparksmith = cardDef("Sparksmith") {
    manaCost = "{1}{R}"
    typeLine = "Creature — Goblin"
    power = 1; toughness = 1

    activatedAbility {
        cost = Costs.Tap
        val t = target("target", Targets.CreatureOrPlayer)
        effect = Effects.DealDamage(
            DynamicAmounts.Count(Player.You, Zone.BATTLEFIELD,
                GameObjectFilter.Creature.withSubtype("Goblin")),
            t
        ).then(Effects.DealDamage(
            DynamicAmounts.Count(Player.You, Zone.BATTLEFIELD,
                GameObjectFilter.Creature.withSubtype("Goblin")),
            EffectTarget.Controller
        ))
    }
}

Static Ability (Lord Effect)

val GoblinKing = cardDef("Goblin King") {
    manaCost = "{1}{R}{R}"
    typeLine = "Creature — Goblin"
    power = 2; toughness = 2

    staticAbility {
        ability = StaticAbilityType.ModifyStatsForCreatureGroup(
            1, 1
        )
        affectsFilter = AffectsFilter.OtherCreaturesWithSubtype("Goblin")
    }
    staticAbility {
        ability = StaticAbilityType.GrantKeywordToGroup(Keyword.MOUNTAINWALK)
        affectsFilter = AffectsFilter.OtherCreaturesWithSubtype("Goblin")
    }
}

Composite Effects (then)

spell {
    val t = target("permanent", Targets.Permanent)
    effect = Effects.ReturnToHand(t)
        .then(Effects.Discard(1))
}

Conditional Effects

triggeredAbility {
    trigger = Triggers.EntersBattlefield
    effect = Effects.Branch(
        condition = Conditions.TopCardMatches(Filters.Land),
        ifTrue = Effects.PutTopCardOnBattlefield,
        ifFalse = Effects.DrawCards(1)
    )
}

Reflexive Triggers

For “When you do” abilities:

triggeredAbility {
    trigger = Triggers.EntersBattlefield
    effect = Effects.ReflexiveTrigger(
        action = Effects.Sacrifice(TargetFilter.Creature.other()),
        optional = true,
        reflexiveEffect = Effects.DealDamage(
            amount = Values.SacrificedCreaturePower,
            target = TargetFilter.Any
        )
    )
}

Library Manipulation (Pipeline Pattern)

Always prefer atomic pipeline effects over monolithic executors. Library/zone mechanics should compose from Gather → Select → Move primitives. 
// Scry 2
effect = Effects.Scry(2)

// Mill 3
effect = Effects.Mill(3)

// Surveil 2 (like scry but graveyard)
effect = Effects.Surveil(2)

// Search library for a basic land
effect = Effects.SearchLibrary(filter = GameObjectFilter.BasicLand)

// Look at top 7, keep 2, rest to graveyard
effect = EffectPatterns.lookAtTopAndKeep(count = 7, keepCount = 2)

// Wheel (each player shuffles hand, draws)
effect = EffectPatterns.wheelEffect(Player.Each)

Morph

val SilentSpecter = cardDef("Silent Specter") {
    manaCost = "{4}{B}{B}"
    typeLine = "Creature — Specter"
    power = 4; toughness = 4
    keywords(Keyword.FLYING)
    morph = "{3}{B}{B}"  // Morph cost

    triggeredAbility {
        trigger = Triggers.DealsCombatDamageToPlayer
        effect = EffectPatterns.discardCards(2, EffectTarget.DamagedPlayer)
    }
}

Aura / Equipment

// Aura
val Pacifism = cardDef("Pacifism") {
    manaCost = "{1}{W}"
    typeLine = "Enchantment — Aura"
    auraTarget = Targets.Creature

    staticAbility {
        ability = StaticAbilityType.PreventAttackAndBlock
        affectsFilter = AffectsFilter.EnchantedCreature
    }
}

// Equipment
val Bonesplitter = cardDef("Bonesplitter") {
    manaCost = "{1}"
    typeLine = "Artifact — Equipment"
    equipAbility("{1}")

    staticAbility {
        ability = StaticAbilityType.ModifyEquippedCreatureStats(2, 0)
        affectsFilter = AffectsFilter.EquippedCreature
    }
}

Registering Cards

// In definitions/{set}/MySet.kt
val MySet = cardSet("MNS", "My New Set") {
    add(LightningBolt)
    add(GlorySeeker)
    add(Sparksmith)
}

// In provider/SetRegistry.kt
object SetRegistry {
    val allSets = listOf(
        PortalSet,
        MySet  // <-- Register here
    )
}

Testing Your Card

Write scenario tests using ScenarioTestBase:

class LightningBoltTest : ScenarioTestBase() {
    init {
        "Lightning Bolt deals 3 damage" {
            val state = setupGame {
                player1 {
                    hand("Lightning Bolt")
                    battlefield("Mountain")
                    library("Mountain")
                }
                player2 {
                    library("Mountain")
                }
            }
            val result = castSpellTargetingPlayer(state, "Lightning Bolt", player2Id)
            result.assertLifeTotal(player2Id, 17)
        }
    }
}

Best Practices

  • Use Effects.* facade methods, never raw effect constructors
  • Prefer EffectPatterns pipelines for library/zone mechanics
  • Use .then() to chain effects, not nested composites
  • Multiple static abilities on one card use multiple staticAbility { } blocks
  • Use TargetFilter for targeting, GameObjectFilter for filtering groups
  • Always give test players at least one card in library to prevent draw-from-empty losses
